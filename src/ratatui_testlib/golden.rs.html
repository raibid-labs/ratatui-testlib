1//! Golden file testing for visual regression detection.
2//!
3//! This module provides golden file (also called "snapshot" or "blessed file") functionality
4//! for TUI applications. Golden files capture the expected terminal output, and tests can
5//! compare current output against these saved baselines to detect visual regressions.
6
7use std::{fs, path::PathBuf, time::SystemTime};

16/// Default directory for golden files.
17const DEFAULT_GOLDEN_DIR: &str = "tests/golden";
18
19/// Header marker for golden file format.
20const GOLDEN_HEADER_START: &str = "--- GOLDEN FILE ---";
21
22/// Content marker for golden file format.
23const GOLDEN_CONTENT_START: &str = "--- CONTENT ---";
24
25/// Get the golden file directory from environment or use default.
26pub fn get_golden_dir() -> PathBuf {
27    std::env::var("GOLDEN_DIR")
28        .map(PathBuf::from)
29        .unwrap_or_else(|_| PathBuf::from(DEFAULT_GOLDEN_DIR))
30}
31
32/// Check if golden files should be updated instead of compared.
33pub fn should_update_goldens() -> bool {
34    std::env::var("UPDATE_GOLDENS")
35        .map(|v| v == "1")
36        .unwrap_or(false)
37}

39/// Metadata for a golden file.
40#[derive(Debug, Clone)]
41pub struct GoldenMetadata {
42    /// Name of the test that created this golden file.
43    pub test_name: String,
44    /// Terminal width in columns.
45    pub width: u16,
46    /// Terminal height in rows.
47    pub height: u16,
48    /// Timestamp when the golden file was created.
49    pub timestamp: String,
50}
51
52impl GoldenMetadata {
53    /// Create new metadata for a golden file.
54    pub fn new(test_name: impl Into<String>, width: u16, height: u16) -> Self {
55        let timestamp = SystemTime::now()
56            .duration_since(SystemTime::UNIX_EPOCH)
57            .map(|d| {
58                let secs = d.as_secs();
59                let dt = chrono::DateTime::<chrono::Utc>::from_timestamp(secs as i64, 0);
60                dt.map(|d| d.format("%Y-%m-%dT%H:%M:%SZ").to_string())
61                    .unwrap_or_else(|| format!("{}s", secs))
62            })
63            .unwrap_or_else(|_| "unknown".to_string());
64
65        Self {
66            test_name: test_name.into(),
67            width,
68            height,
69            timestamp,
70        }
71    }
72
73    /// Serialize metadata to string format for golden file header.
74    pub fn to_header(&self) -> String {
75        format!(
76            "{}\ntest: {}\nsize: {}x{}\ntimestamp: {}\n",
77            GOLDEN_HEADER_START, self.test_name, self.width, self.height, self.timestamp
78        )
79    }
80
81    /// Parse metadata from a golden file header.
82    pub fn from_header(header: &str) -> Option<Self> {
83        let lines: Vec<&str> = header.lines().collect();
84
85        if lines.is_empty() || !lines[0].contains(GOLDEN_HEADER_START) {
86            return None;
87        }
88
89        let mut test_name = String::new();
90        let mut width = 0;
91        let mut height = 0;
92        let mut timestamp = String::new();
93
94        for line in lines.iter().skip(1) {
95            if let Some(value) = line.strip_prefix("test: ") {
96                test_name = value.to_string();
97            } else if let Some(value) = line.strip_prefix("size: ") {
98                if let Some((w, h)) = value.split_once('x') {
99                    width = w.parse().ok()?;
100                    height = h.parse().ok()?;
101                }
102            } else if let Some(value) = line.strip_prefix("timestamp: ") {
103                timestamp = value.to_string();
104            }
105        }
106
107        Some(Self { test_name, width, height, timestamp })
108    }
109}

111/// A golden file containing expected terminal output.
112#[derive(Debug, Clone)]
113pub struct GoldenFile {
114    /// Metadata about the golden file.
115    pub metadata: GoldenMetadata,
116    /// The actual terminal content (with ANSI escape codes preserved).
117    pub content: String,
118}
119
120impl GoldenFile {
121    /// Create a new golden file from screen state.
122    pub fn from_screen_state(test_name: impl Into<String>, state: &ScreenState) -> Self {
123        let (width, height) = state.size();
124        let metadata = GoldenMetadata::new(test_name, width, height);
125        let content = state.contents();
126
127        Self { metadata, content }
128    }
129
130    /// Serialize the golden file to a string.
131    pub fn to_string(&self) -> String {
132        format!("{}{}\n{}", self.metadata.to_header(), GOLDEN_CONTENT_START, self.content)
133    }
134
135    /// Parse a golden file from a string.
136    pub fn from_string(content: &str) -> Result<Self> {
137        let content_start = content.find(GOLDEN_CONTENT_START).ok_or_else(|| {
138            TermTestError::Parse("Golden file missing content marker".to_string())
139        })?;
140
141        let header = &content[..content_start];
142        let content_body = &content[content_start + GOLDEN_CONTENT_START.len()..];
143
144        let metadata = GoldenMetadata::from_header(header).ok_or_else(|| {
145            TermTestError::Parse("Failed to parse golden file header".to_string())
146        })?;
147
148        let content = content_body.trim_start_matches('\n').to_string();
149
150        Ok(Self { metadata, content })
151    }
152
153    /// Save the golden file to disk.
154    pub fn save(&self, name: &str) -> Result<PathBuf> {
155        let golden_dir = get_golden_dir();
156        fs::create_dir_all(&golden_dir)?;
157        let path = golden_dir.join(format!("{}.golden.txt", name));
158        fs::write(&path, self.to_string())?;
159        Ok(path)
160    }
161
162    /// Load a golden file from disk.
163    pub fn load(name: &str) -> Result<Self> {
164        let golden_dir = get_golden_dir();
165        let path = golden_dir.join(format!("{}.golden.txt", name));
166
167        let content = fs::read_to_string(&path).map_err(|e| {
168            TermTestError::Io(std::io::Error::new(
169                e.kind(),
170                format!("Failed to read golden file '{}': {}", path.display(), e),
171            ))
172        })?;
173
174        Self::from_string(&content)
175    }
176
177    /// Compare this golden file against current screen state.
178    pub fn compare(&self, state: &ScreenState) -> Result<()> {
179        let current_content = state.contents();
180
181        if self.content == current_content {
182            return Ok(());
183        }
184
185        let diff = generate_diff(&self.content, &current_content);
186
187        Err(TermTestError::Parse(format!(
188            "Golden file mismatch: {}\n{}",
189            self.metadata.test_name, diff
190        )))
191    }
192}

194/// Generate a unified diff between expected and actual content.
195pub fn generate_diff(expected: &str, actual: &str) -> String {
196    let diff = TextDiff::from_lines(expected, actual);
197
198    let mut output = String::new();
199    output.push_str("--- expected (golden)\n");
200    output.push_str("+++ actual\n");
201
202    let mut line_num = 1;
203
204    for (idx, group) in diff.grouped_ops(3).iter().enumerate() {
205        if idx > 0 {
206            output.push_str("...\n");
207        }
208
209        for op in group {
210            let old_start = op.old_range().start + 1;
211            let old_len = op.old_range().len();
212            let new_start = op.new_range().start + 1;
213            let new_len = op.new_range().len();
214
215            output.push_str(&format!(
216                "@@ -{},{} +{},{} @@\n",
217                old_start, old_len, new_start, new_len
218            ));
219
220            for change in diff.iter_changes(op) {
221                let (sign, style_prefix) = match change.tag() {
222                    ChangeTag::Delete => ("-", "\x1b[31m"),
223                    ChangeTag::Insert => ("+", "\x1b[32m"),
224                    ChangeTag::Equal => (" ", ""),
225                };
226
227                if change.tag() != ChangeTag::Equal {
228                    output.push_str(&format!(
229                        "{}{:>4} {} {}",
230                        style_prefix,
231                        line_num,
232                        sign,
233                        change.value()
234                    ));
235                    if !change.value().ends_with('\n') {
236                        output.push('\n');
237                    }
238                    output.push_str("\x1b[0m");
239                } else {
240                    output.push_str(&format!("{:>4} {} {}", line_num, sign, change.value()));
241                    if !change.value().ends_with('\n') {
242                        output.push('\n');
243                    }
244                }
245
246                line_num += 1;
247            }
248        }
249    }
250
251    output
252}

254/// Save the current screen state as a golden file.
255pub fn save_golden(name: &str, state: &ScreenState) -> Result<PathBuf> {
256    let golden = GoldenFile::from_screen_state(name, state);
257    golden.save(name)
258}
259
260/// Compare current screen state against a golden file.
261pub fn assert_matches_golden(name: &str, state: &ScreenState) -> Result<()> {
262    if should_update_goldens() {
263        let path = save_golden(name, state)?;
264        eprintln!("Updated golden file: {}", path.display());
265        Ok(())
266    } else {
267        let golden = GoldenFile::load(name)?;
268        golden.compare(state)
269    }
270}
271
272/// Update a golden file with new content.
273pub fn update_golden(name: &str, state: &ScreenState) -> Result<PathBuf> {
274    save_golden(name, state)
